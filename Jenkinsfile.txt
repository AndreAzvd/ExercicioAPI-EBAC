pipeline {
    agent any

    stages {

        stage('Checkout') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/AndreAzvd/ExercicioAPI-EBAC'
            }
        }

        stage('Instalar Dependências') {
            steps {
                bat 'npm install'
            }
        }

        stage('Instalar Cypress') {
            steps {
                // Garante que o Cypress está instalado no ambiente do Jenkins
                bat 'npx cypress install'
            }
        }

        stage('Executar Testes') {
            steps {
                bat 'npx cypress run'
            }
        }
    }
}
